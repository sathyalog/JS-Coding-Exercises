Using ES5 

                function removeDups(arrayval) {
                    let unique = {};
                    arrayval.forEach(function(i) {
                      if(!unique[i]) {
                        unique[i] = true;
                      }
                    });
                    return Object.keys(unique);
                }
                  
                const filtered2 = removeDups(arrayval);
